-- 13) 综合客户表(client)、资产表(property)、理财产品表(finances_product)、
--     保险表(insurance)、基金表(fund)和投资资产表(property)，
--     列出所有客户的编号、名称和总资产，总资产命名为total_property。
--     总资产为储蓄卡余额，投资总额，投资总收益的和，再扣除信用卡透支的金额
--     (信用卡余额即为透支金额)。客户总资产包括被冻结的资产。
--    请用一条SQL语句实现该查询：
select c_id, c_name, sum(property1) as total_property
from
(
    select c_id,
        c_name,
        sum(CASE 
            WHEN pro_type = 1 THEN pro_quantity * p_amount 
            WHEN pro_type = 2 THEN pro_quantity * i_amount  
            WHEN pro_type = 3 THEN pro_quantity * f_amount 
            ELSE 0
        END) + COALESCE(sum(pro_income), 0)
        as property1
    from client 
        left join property
        on c_id = pro_c_id
        left join finances_product
        on pro_pif_id = p_id
        left join insurance
        on pro_pif_id = i_id
        left join fund
        on pro_pif_id = f_id
    group by c_id

    union all

    select c_id,
        c_name,
        sum(CASE
            WHEN b_type = '储蓄卡' THEN b_balance
            WHEN b_type = '信用卡' THEN -b_balance
            ELSE 0
        END)
        as property1
    from client 
        left join bank_card
        on c_id = b_c_id
    group by c_id
)
group by c_id, c_name
order by c_id;
